var JOB_ORDER = [ "ضابط قسم", "صف ضابط قسم", "قائد سرية", "قائد مجموعة", "سائق", "عنصر", "أخرى" ]; function doGet(e) { return ContentService.createTextOutput("التطبيق يعمل."); } function doPost(e) { var lock = LockService.getScriptLock(); lock.waitLock(10000); try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); if (sheet.getLastRow() === 0) { sheet.appendRow([ "الاسم الثلاثي", "الرقم العسكري", "المسمى الوظيفي", "المسمى الوظيفي (أخرى)", "بدلة عسكرية", "تاريخ استلام البدلة", "حذاء عسكري", "تاريخ استلام الحذاء", "جعبة درع", "حامل بندقية", "بيريه لون أحمر", "نطاق عسكري", "قفازات" ]); } var params = e.parameter; var fullName = (params.fullName || "").trim(); var militaryId = (params.militaryId || "").trim(); var jobTitle = params.jobTitle || ""; var otherJob = params.otherJob || ""; var suitStatus = params.suitStatus || "لم يستلم"; var suitDate = params.suitDate || ""; if (suitDate && suitDate.indexOf(" ") !== -1) { var parts = suitDate.split(" "); suitDate = parts[0] + "/" + parts[1]; } var shoesStatus = params.shoesStatus || "لم يستلم"; var shoesDate = params.shoesDate || ""; if (shoesDate && shoesDate.indexOf(" ") !== -1) { var parts = shoesDate.split(" "); shoesDate = parts[0] + "/" + parts[1]; } var bag = params.bag || "لم يستلم"; var rifleHolder = params.rifleHolder || "لم يستلم"; var redBeret = params.redBeret || "لم يستلم"; var belt = params.belt || "لم يستلم"; var gloves = params.gloves || "لم يستلم"; // منع تكرار الرقم العسكري var lastRow = sheet.getLastRow(); if (lastRow > 1) { var existingIds = sheet.getRange(2, 2, lastRow - 1, 1).getValues(); for (var i = 0; i < existingIds.length; i++) { if (existingIds[i][0].toString().trim() === militaryId) { return ContentService .createTextOutput(JSON.stringify({ result: "duplicate" })) .setMimeType(ContentService.MimeType.JSON); } } } sheet.appendRow([ fullName, militaryId, jobTitle, (jobTitle === "أخرى") ? otherJob : "", suitStatus, suitDate, shoesStatus, shoesDate, bag, rifleHolder, redBeret, belt, gloves ]); sortSheetByJobTitle(sheet); return ContentService .createTextOutput(JSON.stringify({ result: "success" })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService .createTextOutput(JSON.stringify({ result: "error", message: error.toString() })) .setMimeType(ContentService.MimeType.JSON); } finally { lock.releaseLock(); } } function sortSheetByJobTitle(sheet) { var lastRow = sheet.getLastRow(); if (lastRow <= 1) return; var range = sheet.getRange(2, 1, lastRow - 1, sheet.getLastColumn()); var data = range.getValues(); data.sort(function(rowA, rowB) { var jobA = rowA[2].toString().trim(); var jobB = rowB[2].toString().trim(); var indexA = JOB_ORDER.indexOf(jobA); var indexB = JOB_ORDER.indexOf(jobB); if (indexA === -1) indexA = JOB_ORDER.length; if (indexB === -1) indexB = JOB_ORDER.length; return indexA - indexB; }); range.setValues(data); }
